      ```dart title="Using as a widget"
Stack(
  children: [
    BoxTransform(
      box: box,
      flip: flip,
      onChanged: (box, flip) {
        setState(() {
          this.box = box;
          this.flip = flip;
        });
      },
      contentBuilder: (context, rect, flip) => Transform.scale(
        scaleX: flip.isHorizontal ? -1 : 1,
        scaleY: flip.isVertical ? -1 : 1,
        child: Image.asset(
          'assets/images/landscape.jpg',
          width: rect.width,
          height: rect.height,
          fit: BoxFit.fill,
        ),
      ),
    ),
  ],
);
      ```
